create database bdLivrariaBrasileira
use bdLivrariaBrasileira

create table tbAutor(
idAutor int primary key IDENTITY(1,1)
,nomeAutor varchar(35) not null
)

create table tbGenero(
idGenero int primary key IDENTITY(1,1)
,nomeGenero varchar(19) not null
)

create table tbEditora (
idEditora int primary key IDENTITY(1,1)
,nomeEditora varchar(25) not null
)

create table tbLivro(
idLivro int primary key identity(1,1)
,nomeLivro varchar(58) not null
,numPaginas int not null 
,idGenero int foreign key references tbGenero(idGenero)
,idEditora int foreign key references tbEditora(idEditora)
,idAutor int foreign key references tbAutor(idAutor)
)

//A



select nomeLivro as 'livretes'
from tbLivro 				#O Total de Livros que começam com a letra P
where nomeLivro like 'p%';


//B
select min(numPaginas)as 'menor quantia de páginas ' from tbLivro   # O maior número de páginas entre todos os Livros


//C
select max(numPaginas) as 'maior quantia de páginas' from tbLivro    #O menor Número de páginas entre todos os Livros


//D
select avg(numPaginas) as 'Média de Páginas' from tbLivro    # A média de páginas entre todos os Livros


//E
select sum(numPaginas) as 'total Páginas com a Editora 3' from tbLivro    #A Soma do número de páginas dos Livros de Editora código 01
where idEditora like 1;


//F
select sum(numPaginas) as 'páginas com a letra A de inicio' from tbLivro  #A soma dos números de páginas dos Livros que começãm com a letra A
where nomeLivro like 'A%';

//G
select avg(numPaginas) as 'Média de páginas do autor 3' from tbLivro    #A média dos números de páginas dos Livros que sejam
where idAutor like 3;							#do autor código 03


//H
select sum(1) as 'Livros da Editora 4'			#A quantidade de Livros da Editora de código 04
from tbLivro
where idEditora like 4;


//I
select avg(numPaginas) as 'Média de livros com a palavra estrela no nome' from tbLivro		#A média do número de páginas dos livors que tenham
where nomeLivro like '%estrela%'								# a palavra 'estrela' em seu nome

//J
select sum(1) as 'livros com a palavra poema no nome' from tbLivro			# a quantidade de livros que tenham a palavra 'poema'
where nomeLivro like '%POEMA%'								#em seu nome



//K
select tbGenero.nomeGenero as 'Generos', count(idLivro) as 'livro de cada genero'	#A quantidade de livros agrupado pelo nome do gênero
from tbGenero inner join tbLivro on tbGenero.idGenero = tbLivro.idGenero
group by nomeGenero

//L
select nomeAutor 'Autores' ,sum(numPaginas) as 'Qtd de páginas'				#A soma das págianas agrupada pelo nome do gênero
from tbAutor inner join tbLivro on tbAutor.idAutor = tbLivro.idAutor
group by nomeAutor;



//M
select nomeAutor as 'Autores' , avg(numPaginas) as 'Média de Páginas' 			#A média de páginas agrupada pelo nome do autor em ordem
from tbAutor inner join tbLivro on tbAutor.idAutor = tbLivro.idAutor			#alfabética (de A a Z)
group by nomeAutor 
order by nomeAutor asc;


//N
select nomeEditora as 'Editora' ,count(idLivro) as 'Livros'				#A quantidade de livros agrupada pelo nome da editora
from tbEditora FULL join tbLivro on tbEditora.idEditora = tbLivro.idEditora		#em ordem afabética inversa(de Z a A)
group by nomeEditora
order by nomeEditora DESC;



//O
select nomeAutor as 'Autores' ,sum(numPaginas) as 'Qtd total de Páginas dos livros'	#A soma das páginas dos livros agrupados pelo nome do autor
from tbAutor inner join tbLivro on tbAutor.idAutor = tbLivro.idAutor			#que sejam de autores cujo nome comece com a letra 'C'
where nomeAutor like 'C%'
group by nomeAutor;


											#A quantidade de livros agrupados pelo nome do Autor ,cujo nome do autor seja 'Machado de Assis' ou
//P											#'Cora Coralina' ou 'Graciliano Ramos' ou 'Clarice Lispector'
select nomeAutor as 'Autores' ,count(idLivro) as 'qtd Livros ' 			
from tbAutor full join tbLivro on tbAutor.idAutor = tbLivro.idAutor
where nomeAutor like 'Cora coralina%' or nomeAutor like  'Graciano Ramos%'  or nomeAutor like 'Clarice Lispector% ' or nomeAutor like 'Machado de Assis%'
group by nomeAutor;



//Q  
select nomeEditora as 'Editoras' ,sum(numPaginas) as ' soma qtd Páginas ' 		# A soma das páginas dos livros agrupados pelo nome da Editora cujo número de páginas esteja entre 200 e 500 (inclusive)
from tbEditora inner join tbLivro on tbEditora.idEditora = tbLivro.idEditora
group by nomeEditora
having sum(numPaginas) between 200  and 500;

//R
select nomeAutor as 'Escritor' , tbLivro.nomeLivro as 'Livros ',tbEditora.nomeEditora     #O Nome dos livros ao lado do nome das editoras e do nome dos autores
from tbAutor
inner join tbLivro on tbAutor.idAutor = tbLivro.idAutor inner join tbEditora on tbLivro.idEditora = tbEditora.idEditora



//S										# O nome dos Livros ao lado dos nomes dos autores , somente daqueles cujo nome da editora for 'Cia das Letras'
select nomeLivro as 'Livros' , tbAutor.nomeAutor as 'Autores'    
from tbLivro inner join tbAutor on tbAutor.idAutor = tbLivro.idAutor inner join tbGenero on tbGenero.idGenero = tbLivro.idGenero
where nomeGenero  like 'Cia das letras';

//T
select nomeAutor as 'Autores' , tbLivro.nomeLivro as 'Livros' 			#o nome dos livros ao lado do nome do autor somente dos livros que não forem do autor "Erico Veríssimo"
from tbAutor
inner join tbLivro on tbAutor.idAutor = tbLivro.idAutor
where nomeAutor not like 'Érico Verrísmo';


//U
select nomeAutor as 'Autores' ,nomeLivro as 'Livros'				#Os nomes dos Autores ao lado dos nomes dos Livros , inclusive daqueles que não tem livros cadastrados	
from tbAutor full join tbLivro on tbAutor.idAutor = tbLivro.idAutor



//V
select nomeAutor as 'Autores', tbLivro.nomeLivro as 'Livros' 			#Os nomes dos autores ao lado dos nomes dos livros, inclusive daqueles que não tem autores cadastrados
from tbAutor
full join tbLivro on tbAutor.idAutor = tbLivro.idAutor


//W
select nomeAutor as 'autores' , nomeLivro as 'livrete' 
from tbAutor 
full join tbLivro on tbAutor.idAutor = tbLivro.idLivro 				# o nome dos Autores ao lado dos nomes dos livros , indiferente do autor ter ou não livro publicado 
										# e indiferente do livro pertencer a algum autor
										

//X
select l.nomeEditora , e.nomeLivro						#a editora ática irá publicar todos os titulos dessa livraria , criar um select que associe os nomes  
from tbEditora as l								# de todos os livros ao lado do nome da editora Ática .
full join tbLivro as e
on l.idEditora = e.idLivro
where l.nomeEditora like 'atica';

//Y
 select nomeAutor as 'autores sem livros publicados' 				#Somente os nomes dos autores que não tem livros Cadastrados
 from tbAutor
 full join tbLivro on tbAutor.idAutor = tbLivro.idAutor
 where tbLivro.idAutor is null;


 
//Z
 select nomeGenero as 'Genêros sem Livros cadastrados'				#Os nomes dos Gêneros que não possuem nenhum livro cadastrado
 from tbGenero
 full join tbLivro on tbGenero.idGenero = tbLivro.idGenero
 where tbLivro.idGenero is null;
